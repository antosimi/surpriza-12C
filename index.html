// Variabile globale si vectori de directie (Jos, Dreapta, Sus, Stanga) int n, m, a[10][10], sol[10][10]; int dx[4] = {1, 0, -1, 0}; int dy[4] = {0, 1, 0, -1}; bool valid(int r, int c) { return (r >= 0 && r < n && c >= 0 && c < m && a[r][c] == 0 && sol[r][c] == 0); } void marcheaza_pas(int r, int c) { sol[r][c] = 1; } void sterge_pas(int r, int c) { sol[r][c] = 0; } void afisare_succes() { cout << "Traseu gasit cu succes pentru 12C!"; } void backtracking(int k, int r, int c) { for (int i = 0; i < 4; i++) { int nou_r = r + dx[i]; int nou_c = c + dy[i]; if (valid(nou_r, nou_c)) { marcheaza_pas(nou_r, nou_c); if (nou_r == n - 1 && nou_c == m - 1) { afisare_succes(); } else { backtracking(k + 1, nou_r, nou_c); } sterge_pas(nou_r, nou_c); } } }
Stivă (k) = 1
Direcție (i) = 0
Poziție curentă = [0,0]
